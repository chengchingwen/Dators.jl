# Dators.jl

A data implementation.

```julia
julia> using Dators

julia> src = Dators.CreateSrc() do
           Channel() do ch
               for i = 1:100
                   put!(ch, i)
               end
           end
       end;

julia> dator = Dators.Dator(x->x*x, 1, src;  compute_type=Dators.Thread(2));

julia> bd = Dators.batch(dator, 4);

julia> Dators.start!(bd)
```
